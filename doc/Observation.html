def areas_as_text=(areas_as_text)
  @error_areas =  nil
  new_areas = Area.parse(areas_as_text)
  if String == new_areas.class
    @error_areas = new_areas
  else
    self.areas  = new_areas
  end
end

def in_review=(state)
  return if new_record?
  if state == '0'
    self.publish!
  elsif state == '1'
    self.review!
  end
end
