def put_hazards
   @material = Material.find_by_id(params[:id])
   if @material
     values = params[:hazards].try(:values).to_a # If nil, we get []
     @material.hazards = values.collect { |value| Hazard.find(value) }
     
     redirect_to :action => "edit"
   else
     redirect_to :action => "new"
   end
end

  end

end
